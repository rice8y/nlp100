def cipher(text):
    res = []
    for char in text:
        if 'a' <= char <= 'z':
            new_char = chr(219 - ord(char))
            res.append(new_char)
        else:
            res.append(char)
    return ''.join(res)

text = "Hello, World!"
res = cipher(text)
print(res)

import random

def typoglycemia(text):
    words = text.split()
    for i in range(len(words)):
        if len(words[i]) <= 4:
            continue
        else:
            word = words[i]
            middle = word[1:-1]
            shuffled_middle = ''.join(random.sample(middle, len(middle)))
            words[i] = word[0] + shuffled_middle + word[-1]

    return ' '.join(words)

res = typoglycemia(text)
print(res)
